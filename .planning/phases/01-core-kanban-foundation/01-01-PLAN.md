typedef struct Task {
    char id[37];           // UUID v4
    char title[256];       // Task title
    int completed;         // 0 = unchecked, 1 = checked
    struct Task *next;     // Linked list for column tasks
} Task;

// Column structure
typedef struct Column {
    char name[64];         // Column display name
    Task *tasks;           // Linked list of tasks
    int task_count;        // Number of tasks
} Column;

// Board structure
typedef struct Board {
    Column columns[3];     // To Do, In Progress, Done
    char filename[512];    // Markdown file path
    time_t last_modified;  // For change detection
} Board;
```

Implement Markdown persistence in src/storage.c:

**parse_markdown(Board *board, const char *filepath):**
- Read markdown file line by line
- Detect H2 headers (`## To Do`, `## In Progress`, `## Done`) to identify columns
- Parse `- [ ] task title` for unchecked tasks, `- [x]` for completed
- Build Board structure with linked lists

**write_markdown(Board *board, const char *filepath):**
- Write temp file first: `/tmp/kanban.XXXXXX` using mkstemp()
- Write H2 headers for columns: `## To Do\n`
- Write tasks as `- [ ] Title\n`
- Call fsync() on file descriptor before close
- Use rename() to atomically replace original

**board_load(Board *board, const char *filepath):**
- Call parse_markdown to populate board
- Handle file not found (create empty board)
- Return 0 on success, -1 on error

**board_save(Board *board, const char *filepath):**
- Call write_markdown with atomic write pattern
- Auto-save triggered by create/delete/move operations (per D-14)

Default file location: `~/.config/kanban-cli/boards/default.md`

#include <ncursesw/ncurses.h>
#include <signal.h>
#include "kanban.h"
#include "storage.h"

static Board global_board;

// Signal handlers for graceful exit
static void handle_signal(int sig) {
    endwin();
    exit(0);
}

// Cleanup function registered with atexit()
static void cleanup(void) {
    endwin();
}

int main(int argc, char *argv[]) {
    // Register cleanup handlers
    signal(SIGINT, handle_signal);
    signal(SIGTERM, handle_signal);
    atexit(cleanup);
    
    // Initialize ncurses
    initscr();
    cbreak();
    noecho();
    keypad(stdscr, TRUE);
    
    // Reduce escape key delay for vim feel
    set_escdelay(25);
    
    // Load or create board
    board_init(&global_board);
    char *home = getenv("HOME");
    snprintf(global_board.filename, sizeof(global_board.filename),
             "%s/.config/kanban-cli/boards/default.md", home);
    board_load(&global_board, global_board.filename);
    
    // Placeholder: render board placeholder, handle 'q' to quit
    // Phase 2 will add full rendering and input handling
    
    endwin();
    return 0;
}
```
