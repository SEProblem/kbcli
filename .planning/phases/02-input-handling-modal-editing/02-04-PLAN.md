```c
// Resize handling pattern
if (ch == KEY_RESIZE) {
    endwin();
    refresh();
    // Re-create or resize windows here
    clear();
    redraw_all_windows();
}
```

## Key Implementation Notes

Per PITFALLS.md, use resizeterm() not full reinitialization:
- KEY_RESIZE is sent when terminal is resized
- LINES and COLS globals update after resizeterm()
- Must recalculate window positions
- Must redraw all windows after resize

```c
// Recalculate window positions based on new dimensions
void renderer_calculate_layout(Renderer* r, int lines, int cols) {
    int col_width = cols / 3;
    int header_height = 2;
    int col_height = lines - header_height - 1;  // -1 for status bar
    
    // Update column windows
    for (int i = 0; i < 3; i++) {
        WINDOW* win = r->column_windows[i];
        if (win) {
            delwin(win);
        }
        r->column_windows[i] = newwin(col_height, col_width, header_height, i * col_width);
    }
    
    // Update status bar
    if (r->status_bar) {
        delwin(r->status_bar);
    }
    r->status_bar = newwin(1, cols, lines - 1, 0);
}

// Redraw all windows
void renderer_redraw_all(Renderer* r) {
    clear();  // Clear virtual screen
    
    // Redraw all columns
    for (int i = 0; i < 3; i++) {
        render_column(r->column_windows[i], r->board->columns + i, i == r->selection.current_column);
    }
    
    // Redraw status bar
    render_status_bar(r->status_bar, r->app);
    
    // Update panels and refresh
    update_panels();
    doupdate();
}
```

Also add helper function to track current renderer state.
