void handle_mouse_event(Application* app) {
    MEVENT event;
    if (getmouse(&event) == OK) {
        // Check for button press
        if (event.bstate & BUTTON1_PRESSED || 
            event.bstate & BUTTON1_CLICKED) {
            // event.y, event.x are screen coordinates
            navigation_select_task_at(app, event.y, event.x);
        }
        // Handle scroll
        if (event.bstate & BUTTON4_PRESSED) {
            // Scroll up
            navigation_scroll_up(app);
        }
        if (event.bstate & BUTTON5_PRESSED) {
            // Scroll down  
            navigation_scroll_down(app);
        }
    }
}
```

void navigation_select_task_at(Application* app, int screen_y, int screen_x) {
    // Get column positions from renderer
    // For each column, check if screen_y/screen_x falls within its bounds
    // Then find the specific task based on row within column
    
    Board* board = app->board;
    Selection* sel = &app->selection;
    
    // Calculate which column was clicked
    int col_width = COLS / 3;
    int col = screen_x / col_width;
    if (col > 2) col = 2;
    
    // Get the column
    Column* column = &board->columns[col];
    
    // Calculate which task in the column
    int header_height = 2;  // Column header rows
    int task_height = 1;    // Each task is 1 row
    int task_idx = (screen_y - header_height) / task_height;
    
    // Clamp to valid range
    if (task_idx < 0) task_idx = 0;
    if (task_idx >= column->task_count) task_idx = column->task_count - 1;
    
    // Update selection
    sel->current_column = col;
    sel->task_index = task_idx;
    sel->selected_task = column->tasks[task_idx];
}
```

Also implement scroll functions:
```c
void navigation_scroll_up(Application* app) {
    app->selection.task_index = max(0, app->selection.task_index - 3);
}

void navigation_scroll_down(Application* app) {
    Column* col = &app->board->columns[app->selection.current_column];
    app->selection.task_index = min(col->task_count - 1, app->selection.task_index + 3);
}
```
